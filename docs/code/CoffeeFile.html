   def load filename
      if filename.nil?
         @fname = ""
      else
         @fname = filename
      end

      if @fname.empty?
         log "Empty filename, could not load"
      else
         file = File.new(@fname, 'r')
         @text = file.read
         @changed = true
         file.close
         GlobalSettings.instance.text.text = @text
      end

      return @text
   end
