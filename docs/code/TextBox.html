class TextBox < Qt::Widget
   def initialize
      super

      @gv = Qt::GraphicsView.new
      scene = Qt::GraphicsScene.new
      @tb = Qt::GraphicsTextItem.new
      @tb.setTextInteractionFlags Qt::TextEditable
      @tb.setTextWidth 550
      @tb1 = Qt::GraphicsTextItem.new
      @tb1.setTextInteractionFlags Qt::TextEditable
      @tb1.setTextWidth 550
      scene.addItem @tb
      scene.addItem @tb1
      scene.setFocusItem @tb
      @gv.setScene scene

      layout = Qt::VBoxLayout.new()
      layout.addWidget(@gv)
      setLayout(layout)
   end

   def text= txt
      @tb.setPlainText txt
   end
end
